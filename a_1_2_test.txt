package com.example.dashboard.web;

import com.example.dashboard.importrequest.ImportRequest;
import com.example.dashboard.importrequest.ImportRequestService;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.multipart.MultipartFile;

import java.util.ArrayList;
import java.util.List;

@Controller
public class ImportController {

    private final ImportRequestService importRequestService;

    public ImportController(ImportRequestService importRequestService) {
        this.importRequestService = importRequestService;
    }

    @GetMapping("/import")
    public String form() {
        return "import";
    }

    @PostMapping("/import")
    public String handleUpload(
            @RequestParam("file") MultipartFile file,
            @RequestParam(value = "userId", required = false) String userId,
            @RequestParam(value = "environment", required = false) String environment,
            @RequestParam(value = "count", required = false) String countRaw,
            Model model) {

        List<String> errors = new ArrayList<>();
        String savedFilePath = null;

        if (userId == null || userId.isBlank()) {
            errors.add("User ID is required");
        }
        if (environment == null || environment.isBlank()) {
            errors.add("Environment is required");
        }
        int count = 0;
        if (countRaw == null || countRaw.isBlank()) {
            errors.add("Count is required");
        } else {
            try {
                count = Integer.parseInt(countRaw.trim());
                if (count < 0) {
                    errors.add("Count must be zero or positive");
                }
            } catch (NumberFormatException ex) {
                errors.add("Count must be a valid integer");
            }
        }

        if (!errors.isEmpty()) {
            model.addAttribute("errors", errors);
            return "import";
        }

        try {
            ImportRequest savedRequest = importRequestService.submit(file, userId.trim(), environment.trim(), count);
            savedFilePath = savedRequest.getFilePath();
            model.addAttribute("requestId", savedRequest.getId());
            model.addAttribute("status", savedRequest.getStatus());
            model.addAttribute("savedFilename", savedRequest.getFilename());
        } catch (IllegalArgumentException ex) {
            errors.add(ex.getMessage());
        } catch (Exception ex) {
            errors.add("Failed to process upload: " + ex.getMessage());
        }

        model.addAttribute("errors", errors);
        if (savedFilePath != null && errors.isEmpty()) {
            model.addAttribute("message", "File saved successfully.");
        }
        return "import";
    }
}
